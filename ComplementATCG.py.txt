#program can complement but does not spit out error message as expected...it still ends program
import re

def complement(sequence):
        complement ={'A' : 'T' , 'T' : 'A' , 'G' : 'C', 'C':'G'}
        comp = ""
        for nt in sequence:
                comp += complement.get(nt)
        return comp

def main():
        right_seq =True



        #sequence = "ACTGAAGCGCTG"


        while (right_seq):

                sequence = input("Please enter a sequence to get complement: \n")
                m =re.match(r'[ATCG]+$' ,sequence.strip())

                if m:
                        new_seq = complement(sequence)
                        print(new_seq)

                #if the letter is not A, T, C, G set right_seq = False
                else:
                        break

        print("Incorrect sequence")



if __name__ == '__main__':
        main()
